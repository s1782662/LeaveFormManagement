150:     /**
151:      * 
152:      * Fetches the unified constructor params and setter values for a class.
153:      * 
154:      * @param string $class The class name to fetch values for.
155:      * 
156:      * @return array An array with two elements; 0 is the constructor values 
157:      * for the class, and 1 is the setter methods and values for the class.
158:      * 
159:      */
160:     public function fetch($class)
161:     {
162:         // have values already been unified for this class?
163:         if (isset($this->unified[$class])) {
164:             return $this->unified[$class];
165:         }
166: 
167:         // fetch the values for parents so we can inherit them
168:         $pclass = get_parent_class($class);
169:         if ($pclass) {
170:             // parent class values
171:             list($parent_params, $parent_setter) = $this->fetch($pclass);
172:         } else {
173:             // no more parents; get top-level values for all classes
174:             $parent_params = $this->params['*'];
175:             $parent_setter = $this->setter['*'];
176:         }
177: 
178:         // stores the unified config and setter values
179:         $unified_params = [];
180:         $unified_setter = [];
181: 
182:         // reflect on the class
183:         $rclass = $this->getReflect($class);
184: 
185:         // does it have a constructor?
186:         $rctor = $rclass->getConstructor();
187:         if ($rctor) {
188:             // reflect on what params to pass, in which order
189:             $params = $rctor->getParameters();
190:             foreach ($params as $param) {
191:                 $name = $param->name;
192:                 $explicit = $this->params->offsetExists($class)
193:                          && isset($this->params[$class][$name]);
194:                 if ($explicit) {
195:                     // use the explicit value for this class
196:                     $unified_params[$name] = $this->params[$class][$name];
197:                 } elseif (isset($parent_params[$name])) {
198:                     // use the implicit value for the parent class
199:                     $unified_params[$name] = $parent_params[$name];
200:                 } elseif ($param->isDefaultValueAvailable()) {
201:                     // use the external value from the constructor
202:                     $unified_params[$name] = $param->getDefaultValue();
203:                 } else {
204:                     // no value, use a null placeholder
205:                     $unified_params[$name] = null;
206:                 }
207:             }
208:         }
209: 
210:         // merge the setters
211:         if (isset($this->setter[$class])) {
212:             $unified_setter = array_merge($parent_setter, $this->setter[$class]);
213:         } else {
214:             $unified_setter = $parent_setter;
215:         }
216: 
217:         // look for setters inside traits
218:         $uses = class_uses($class);
219:         foreach ($uses as $use) {
220:             if (isset($this->setter[$use])) {
221:                 $unified_setter = array_merge($this->setter[$use], $unified_setter);
222:             }
223:         }
224: 
225:         // done, return the unified values
226:         $this->unified[$class][0] = $unified_params;
227:         $this->unified[$class][1] = $unified_setter;
228:         return $this->unified[$class];
229:     }
