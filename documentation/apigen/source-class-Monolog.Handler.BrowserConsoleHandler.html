 17: /**
 18:  * Handler sending logs to browser's javascript console with no browser extension required
 19:  *
 20:  * @author Olivier Poitrey <rs@dailymotion.com>
 21:  */
 22: class BrowserConsoleHandler extends AbstractProcessingHandler
 23: {
 24:     protected static $initialized = false;
 25:     protected static $records = array();
 26: 
 27:     /**
 28:      * {@inheritDoc}
 29:      *
 30:      * Formatted output may contain some formatting markers to be transfered to `console.log` using the %c format.
 31:      *
 32:      * Example of formatted string:
 33:      *
 34:      *     You can do [[blue text]]{color: blue} or [[green background]]{background-color: green; color: white}
 35:      *
 36:      */
 37:     protected function getDefaultFormatter()
 38:     {
 39:         return new LineFormatter('[[%channel%]]{macro: autolabel} [[%level_name%]]{font-weight: bold} %message%');
 40:     }
 41: 
 42:     /**
 43:      * {@inheritDoc}
 44:      */
 45:     protected function write(array $record)
 46:     {
 47:         // Accumulate records
 48:         self::$records[] = $record;
 49: 
 50:         // Register shutdown handler if not already done
 51:         if (PHP_SAPI !== 'cli' && !self::$initialized) {
 52:             self::$initialized = true;
 53:             register_shutdown_function(array('Monolog\Handler\BrowserConsoleHandler', 'send'));
 54:         }
 55:     }
 56: 
 57:     /**
 58:      * Convert records to javascript console commands and send it to the browser.
 59:      * This method is automatically called on PHP shutdown if output is HTML.
 60:      */
 61:     public static function send()
 62:     {
 63:         // Check content type
 64:         foreach (headers_list() as $header) {
 65:             if (stripos($header, 'content-type:') === 0) {
 66:                 if (stripos($header, 'text/html') === false) {
 67:                     // This handler only works with HTML outputs
 68:                     return;
 69:                 }
 70:                 break;
 71:             }
 72:         }
 73: 
 74:         if (count(self::$records)) {
 75:             echo '<script>' . self::generateScript() . '</script>';
 76:             self::reset();
 77:         }
 78:     }
 79: 
 80:     /**
 81:      * Forget all logged records
 82:      */
 83:     public static function reset()
 84:     {
 85:         self::$records = array();
 86:     }
 87: 
 88:     private static function generateScript()
 89:     {
 90:         $script = array();
 91:         foreach (self::$records as $record) {
 92:             $context = self::dump('Context', $record['context']);
 93:             $extra = self::dump('Extra', $record['extra']);
 94: 
 95:             if (empty($context) && empty($extra)) {
 96:                 $script[] = self::call_array('log', self::handleStyles($record['formatted']));
 97:             } else {
 98:                 $script = array_merge($script,
 99:                     array(self::call_array('groupCollapsed', self::handleStyles($record['formatted']))),
100:                     $context,
101:                     $extra,
102:                     array(self::call('groupEnd'))
103:                 );
104:             }
105:         }
106:         return "(function(c){if (c && c.groupCollapsed) {\n" . implode("\n", $script) . "\n}})(console);";
107:     }
108: 
109:     private static function handleStyles($formatted)
110:     {
111:         $args = array(self::quote('font-weight: normal'));
112:         $format = '%c' . $formatted;
113:         preg_match_all('/\[\[(.*?)\]\]\{([^}]*)\}/s', $format, $matches, PREG_OFFSET_CAPTURE | PREG_SET_ORDER);
114: 
115:         foreach (array_reverse($matches) as $match) {
116:             $args[] = self::quote(self::handleCustomStyles($match[2][0], $match[1][0]));
117:             $args[] = '"font-weight: normal"';
118: 
119:             $pos = $match[0][1];
120:             $format = substr($format, 0, $pos) . '%c' . $match[1][0] . '%c' . substr($format, $pos + strlen($match[0][0]));
121:         }
122: 
123:         array_unshift($args, self::quote($format));
124: 
125:         return $args;
126:     }
127: 
128:     private static function handleCustomStyles($style, $string)
129:     {
130:         static $colors = array('blue', 'green', 'red', 'magenta', 'orange', 'black', 'grey');
131:         static $labels = array();
132: 
133:         return preg_replace_callback('/macro\s*:(.*?)(?:;|$)/', function($m) use($string, &$colors, &$labels) {
134:             if (trim($m[1]) === 'autolabel') {
135:                 // Format the string as a label with consistent auto assigned background color
136:                 if (!isset($labels[$string])) {
137:                     $labels[$string] = $colors[count($labels) % count($colors)];
138:                 }
139:                 $color = $labels[$string];
140: 
141:                 return "background-color: $color; color: white; border-radius: 3px; padding: 0 2px 0 2px";
142:             }
143: 
144:             return $m[1];
145:         }, $style);
146:     }
147: 
148:     private static function dump($title, array $dict)
149:     {
150:         $script = array();
151:         $dict = array_filter($dict);
152:         if (empty($dict)) {
153:             return $script;
154:         }
155:         $script[] = self::call('log', self::quote('%c%s'), self::quote('font-weight: bold'), self::quote($title));
156:         foreach ($dict as $key => $value) {
157:             $value = json_encode($value);
158:             if (empty($value)) {
159:                 $value = self::quote('');
160:             }
161:             $script[] = self::call('log', self::quote('%s: %o'), self::quote($key), $value);
162:         }
163:         return $script;
164:     }
165: 
166:     private static function quote($arg)
167:     {
168:         return '"' . addcslashes($arg, "\"\n") . '"';
169:     }
170: 
171:     private static function call()
172:     {
173:         $args = func_get_args();
174:         $method = array_shift($args);
175:         return self::call_array($method, $args);
176:     }
177: 
178:     private static function call_array($method, array $args)
179:     {
180:         return 'c.' . $method . '(' . implode(', ', $args) . ');';
181:     }
